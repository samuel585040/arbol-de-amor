// Fecha de aniversario base const aniversario = new Date("2022-02-13"); // Elemento donde se mostrará el conteo const countdownElement = document.getElementById("countdown"); function actualizarCountdown() { const hoy = new Date(); // Calcular el próximo aniversario let proximo = new Date(hoy.getFullYear(), 1, 13); // febrero = 1 if (hoy > proximo) { proximo = new Date(hoy.getFullYear() + 1, 1, 13); } // Calcular diferencia en días const diffTime = proximo - hoy; const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24)); // Mostrar mensaje countdownElement.innerHTML = `

Fecha de aniversario: 13 de febrero de 2022

Días que faltan para el próximo aniversario: ${diffDays} días

`; } // Actualizar inmediatamente y luego cada día actualizarCountdown(); setInterval(actualizarCountdown, 1000 * 60 * 60); // cada hora
